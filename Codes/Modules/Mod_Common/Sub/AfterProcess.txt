Public Sub AfterProcess(ByVal calculation As XlCalculation, Optional ByRef excelApp As Excel.Application, Optional ByVal isWbOpening As Boolean)

    '引数 既定値判定
    If excelApp Is Nothing Then Set excelApp = Excel.Application

    '各プロセス 再開
    With excelApp
        If .Workbooks.Count > 0 Then
            .Calculation = calculation
        End If
        If Not isWbOpening Then .CutCopyMode = False
        .ScreenUpdating = True
        .DisplayAlerts = True
        .EnableEvents = True
        .StatusBar = False
    End With

End Sub
