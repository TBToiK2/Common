Public Function GetFileName(ByVal FileSpec As String) As String
On Error GoTo Err

    'ファイル名 取得
    Dim FSO As New FileSystemObject
    GetFileName = FSO.GetFileName(FileSpec)

    Exit Function

'エラー処理
Err:

    Call ShowErrMsg(Err.Description, Err.Number)

End Function
