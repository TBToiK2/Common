Public Function BuildPath(ParamArray Path()) As String
On Error GoTo Err

    'パス 作成
    Dim FSO As New FileSystemObject

    Select Case UBound(Path, 1)

        Case 0
            BuildPath = FSO.BuildPath(Path(0), vbNullString)

    Exit Function

'エラー処理
Err:

    Call ShowErrMsg(Err.Description, Err.Number)

End Function
