Public Function GetPath(ByVal spec As String) As String
On Error GoTo Err

    'パス 取得
    Dim FSO As New FileSystemObject
    GetPath = FSO.GetParentFolderName(spec)

    Exit Function

'エラー処理
Err:

    Call ShowErrMsg(Err.Description, Err.Number, "GetPath")

End Function
