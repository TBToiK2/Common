Public Function HasList(ByVal listName As String, Optional ByVal parent As ListObjects) As Boolean
On Error GoTo Err

    '引数 既定値判定
    If IsBlank(parent) Then Set parent = ThisWorkbook.ActiveSheet.ListObjects

    '全リスト 確認
    Dim list As ListObject
    For Each list In parent
        If list.Name = listName Then
            HasList = True
            Exit Function
        End If
    Next list

    Exit Function

'エラー処理
Err:

    Call ShowErrMsg(Err.Description, Err.Number, "HasList")

End Function
