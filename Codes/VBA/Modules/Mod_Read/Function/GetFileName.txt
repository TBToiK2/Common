Public Function GetFileName(ByVal fileSpec As String, Optional ByVal fileType As ReFileType) As String
On Error GoTo Err

    Dim FSO As New FileSystemObject
    With FSO

        'ファイル名 取得
        Select Case searchFileType
            Case reFileName
            GetFileName = .GetFileName(fileSpec)

            Case reBaseName
            GetFileName = .GetBaseName(fileSpec)

            Case reExtensionName
            GetFileName = .GetExtensionName(fileSpec)

        End Select

    End With

    Exit Function

'エラー処理
Err:

    Call ShowErrMsg(Err.Description, Err.Number, "GetFileName")

End Function
