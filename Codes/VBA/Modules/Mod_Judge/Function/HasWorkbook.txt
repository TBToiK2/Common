Public Function HasWorkbook(ByVal wbName As String, Optional ByVal Parent As Workbooks) As Boolean
On Error GoTo Err

    '引数 規定値設定
    If Parent Is Nothing Then Parent = ThisWorkbook

    '全ワークブック 確認
    Dim wb As Workbook
    For Each wb In Parent
        If wb.Name = wbName Then
            HasWorkbook = True
            Exit Function
        End If
    Next wb

    Exit Function

'エラー処理
Err:

    Call ShowErrMsg(Err.Description, Err.Number)

End Function
