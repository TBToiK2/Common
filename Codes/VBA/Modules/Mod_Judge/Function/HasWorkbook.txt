Public Function HasWorkbook(ByVal Parent As Workbooks, ByVal wbName As String) As Boolean
On Error GoTo Err

    '全ワークブック 確認
    Dim wb As Workbook
    For Each wb In Parent
        If wb.Name = wbName Then
            HasWorkbook = True
            Exit Function
        End If
    Next wb

    Exit Function

'エラー処理
Err:

    Call ShowErrMsg(Err.Description, Err.Number)

End Function
