Public Function HasList(ByVal Parent As ListObjects, ByVal ListName As String) As Boolean
On Error GoTo Err

    '全リスト 確認
    Dim List As ListObject
    For Each List In Parent
        If List.Name = ListName Then
            HasList = True
            Exit Function
        End If
    Next List

    Exit Function

'エラー処理
Err:

    Call ShowErrMsg(Err.Description, Err.Number)

End Function
