Public Function HasList(ByVal ListName As String, Optional ByVal Parent As ListObjects) As Boolean
On Error GoTo Err

    '引数 既定値判定
    If IsBlank(Parent) Then Parent = ActiveSheet.ListObjects

    '全リスト 確認
    Dim List As ListObject
    For Each List In Parent
        If List.Name = ListName Then
            HasList = True
            Exit Function
        End If
    Next List

    Exit Function

'エラー処理
Err:

    Call ShowErrMsg(Err.Description, Err.Number)

End Function
