Public Function IsOpen(ByVal FileSpec As String) As Boolean
On Error Resume Next

    Dim FileNumber As LongPtr
    FileNumber = FreeFile

    '開閉確認
    Open FileSpec For Append As #CLng(FileNumber)
    Close #CLng(FileNumber)

    IsOpen = Err.Number > 0

End Function
