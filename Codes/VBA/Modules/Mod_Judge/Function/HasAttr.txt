Public Function HasAttr(ByVal spec As String, ByVal attr As FileAttribute) As Boolean
On Error GoTo Err

    Dim FSO As New FileSystemObject

    '属性 比較(ビット演算)
    If GetAttr(spec) And vbDirectory = vbDirectory Then

        'フォルダ属性 比較(ビット演算)
        HasAttr = (FSO.GetFolder(spec).Attributes And attr) = attr

    Else

        'ファイル属性 比較(ビット演算)
        HasAttr = (FSO.GetFile(spec).Attributes And attr) = attr

    End If

    Exit Function

'エラー処理
Err:

    Call ShowErrMsg(Err.Description, Err.Number)

End Function
