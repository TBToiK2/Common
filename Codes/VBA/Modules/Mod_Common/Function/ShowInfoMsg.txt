Public Function ShowInfoMsg(ByVal prompt As String, Optional ByVal title As String) As LongPtr
On Error Resume Next

    'タイトル 設定
    If Not IsBlank(title) Then title = ":" & title

    'メッセージ 表示
    ShowInfoMsg = MsgBox(prompt, vbOKOnly + vbInformation, "情報" & title)

End Function
