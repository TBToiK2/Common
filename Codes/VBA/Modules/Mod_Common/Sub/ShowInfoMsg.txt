Public Sub ShowInfoMsg(ByVal prompt As String, Optional ByVal title As String)
On Error Resume Next

    'タイトル 設定
    If Not IsBlank(title) Then title = ":" & title

    'メッセージ 表示
    Call MsgBox(prompt, vbOKOnly + vbInformation, "情報" & title)

End Sub
