Public Sub ShowAllData(ByVal ws As Worksheet)

    With ws
        'フィルター 解除
        If .FilterMode Then Call .ShowAllData

        '行列 全表示
        .Cells.EntireColumn.Hidden = False
        .Cells.EntireRow.Hidden = False
    End With

End Sub
