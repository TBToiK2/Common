Public Sub AfterProcess(ByVal Calculation As XlCalculation, Optional ByRef ExcelApp As Application)

    '引数 既定値判定
    If IsBlank(ExcelApp) Then Set ExcelApp = Application

    '各プロセス 再開
    With ExcelApp
        If .Workbooks.Count > 0 Then
            .Calculation = Calculation
        End If
        .CutCopyMode = False
        .ScreenUpdating = True
        .DisplayAlerts = True
        .EnableEvents = True
    End With
    DoEvents

End Sub
